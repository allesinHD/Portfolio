function metrics(){
  const p = S.positions;
  const totalPos = p.filter(e=>{const c=CAT(e.cat_id);return c&&!c.neg;}).reduce((s,e)=>s+pv(e),0);
  const totalNeg = p.filter(e=>{const c=CAT(e.cat_id);return c&&c.neg;}).reduce((s,e)=>s+parseFloat(e.outstanding||0),0);
  const net = totalPos - totalNeg;
  const liq = p.filter(e=>['girokonto','tagesgeld','festgeld','kaution'].includes(e.cat_id)).reduce((s,e)=>s+pv(e),0);
  const sec = p.filter(e=>['etf','aktien','robo','krypto'].includes(e.cat_id)).reduce((s,e)=>s+pv(e),0);
  const krypto = p.filter(e=>e.cat_id==='krypto').reduce((s,e)=>s+pv(e),0);
  const sparraten = p.filter(e=>e.sparrate).reduce((s,e)=>s+parseFloat(e.sparrate||0),0);
  const ytdGain = p.filter(e=>e.ytd).reduce((s,e)=>s+pv(e)*(parseFloat(e.ytd||0)/100),0);
  const ytdPct = net>0?(ytdGain/(net-ytdGain))*100:0;
  const cf = S.cashflow; const netCF=(cf.inc||0)-(cf.exp||0);
  const byLabel={};
  p.forEach(e=>{const c=CAT(e.cat_id);if(!c||c.neg)return;byLabel[c.label]=(byLabel[c.label]||0)+pv(e);});
  const equityPct=(((byLabel['ETF / Fonds']||0)+(byLabel['Aktien Einzeltitel']||0)+(byLabel['Robo-Advisory']||0))/net)*100||0;
  const liqPct=(liq/net)*100||0;
  const kryptoPct=(krypto/net)*100||0;
  let risk=4;
  if(kryptoPct>10)risk++;if(equityPct>80)risk++;if(liqPct<5)risk++;
  risk=Math.min(risk,9);
  return {net,totalPos,totalNeg,liq,sec,krypto,sparraten,ytdGain,ytdPct,netCF,byLabel,equityPct,liqPct,kryptoPct,risk,p};
}

function donutSVG(data, total, size=160){
  if(!data.length||!total) return `<svg width="${size}" height="${size}"><circle cx="${size/2}" cy="${size/2}" r="${size/2-14}" fill="none" stroke="#232c3d" stroke-width="20"/><text x="${size/2}" y="${size/2+5}" text-anchor="middle" fill="#4a5568" font-size="11" font-family="monospace">Leer</text></svg>`;
  const cx=size/2,cy=size/2,r=size/2-14,ri=r*.62;
  let angle=-Math.PI/2,paths='';

    const a1=angle,a2=angle+frac*2*Math.PI; angle=a2;
    const x1=cx+r*Math.cos(a1),y1=cy+r*Math.sin(a1);
    const x2=cx+r*Math.cos(a2),y2=cy+r*Math.sin(a2);
    const xi1=cx+ri*Math.cos(a1),yi1=cy+ri*Math.sin(a1);
    const xi2=cx+ri*Math.cos(a2),yi2=cy+ri*Math.sin(a2);
    const lg=frac>.5?1:0;
    paths+=`<path d="M${xi1},${yi1} L${x1},${y1} A${r},${r} 0 ${lg} 1 ${x2},${y2} L${xi2},${yi2} A${ri},${ri} 0 ${lg} 0 1 ${xi1},${yi1}Z" fill="${d.col}" opacity=".9" stroke="#0d1117" stroke-width="1.5"/>`;
  });
  return `<svg width="${size}" height="${size}">${paths}<circle cx="${cx}" cy="${cy}" r="${ri-2}" fill="#161b27"/><text x="${cx}" y="${cy-6}" text-anchor="middle" fill="#e2e8f0" font-size="12" font-family="monospace" font-weight="500">${eur(total,0)}</text><text x="${cx}" y="${cy+10}" text-anchor="middle" fill="#8899aa" font-size="9" font-family="monospace">Netto</text></svg>`;
}

function sparkSVG(pts,color,w=600,h=140){
  if(!pts||pts.length<2) return `<div style="height:${h}px;display:flex;align-items:center;justify-content:center;background:var(--panel2);border-radius:8px;color:var(--textd);font-size:11px;font-family:monospace;">Positionen erfassen → Tab „Positionen"</div>`;
  const mn=Math.min(...pts)*.98,mx=Math.max(...pts)*1.01,rng=mx-mn||1;
  const n=pts.length;
  const xs=pts.map((_,i)=>i/(n-1)*w);
  const ys=pts.map(v=>h-(v-mn)/rng*(h-8)-4);
  const d='M'+xs.map((x,i)=>`${x},${ys[i]}`).join(' L');
  return `<svg width="100%" height="${h}" viewBox="0 0 ${w} ${h}" preserveAspectRatio="none" style="display:block">
    <defs><linearGradient id="sg" x1="0" y1="0" x2="0" y2="1"><stop offset="0%" stop-color="${color}" stop-opacity=".25"/><stop offset="100%" stop-color="${color}" stop-opacity="0"/></linearGradient></defs>
    <path d="${d} L${w},${h} L0,${h} Z" fill="url(#sg)"/>
    <path d="${d}" fill="none" stroke="${color}" stroke-width="2" stroke-linejoin="round"/>
    <circle cx="${xs[xs.length-1]}" cy="${ys[ys.length-1]}" r="4" fill="${color}"/>
  </svg>`;
}

function riskGauge(score){
  const col=score<=4?'#22c55e':score<=6?'#f59e0b':'#ef4444';
  const lbl=score<=4?'Konservativ':score<=6?'Moderat':'Aggressiv';
  const arc=score/10*125.6;
  return `<div style="text-align:center">
    <svg width="110" height="60" viewBox="0 0 110 60">
      <path d="M12,54 A43,43 0 0 1 98,54" fill="none" stroke="#232c3d" stroke-width="9" stroke-linecap="round"/>
      <path d="M12,54 A43,43 0 0 1 98,54" fill="none" stroke="${col}" stroke-width="9" stroke-linecap="round" stroke-dasharray="${arc} 135" opacity=".9"/>
      <text x="55" y="50" text-anchor="middle" fill="${col}" font-size="22" font-family="monospace" font-weight="600">${score}</text>
    </svg>
    <div style="font-family:monospace;font-size:10px;color:${col};margin-top:-4px">${lbl}</div>
  </div>`;
}

function progBar(val,max,col,thick=false){
  const p=Math.min((val/max)*100,100)||0;
  return `<div class="bar-track${thick?' bar-thick':''}"><div class="bar-fill" style="width:${p}%;background:${col}"></div></div>`;
}
